EX:
    console.log("Hello Word");

let = é uma variavel que eu mudar o valor dela.
EX:
    let variavel = 10;

const = é uma variavel que não pode mudar o valor dela. 
EX:
    const pi = 3.14;

Math.pow = faz parte da biblioteca de funções matematicas
EX:
    Math.pow(altura,2);
